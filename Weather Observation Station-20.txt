DECLARE @N INT = (((SELECT COUNT(ID) FROM STATION)+1)/2);  
CREATE TABLE #TEMP
(
    ID INT IDENTITY(1,1),
    CITY VARCHAR(21), 
    STATE VARCHAR(2), 
    LAT_N decimal(18,4),
    LONG_W decimal(18,4)
);
INSERT INTO #TEMP(CITY,STATE,LAT_N,LONG_W) 
          SELECT CITY,STATE,LAT_N,LONG_W FROM STATION ORDER BY LAT_N;
SELECT LAT_N FROM #TEMP WHERE ID = @N
DROP TABLE #TEMP
